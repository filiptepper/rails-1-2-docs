def test_date_select_with_different_order

  @post.written_on = Date.new(2004, 6, 15)

  expected =  %Q{<select id="post_written_on_3i" name="post[written_on(3i)]">\n}

  expected <<  %Q{<select id="post_written_on_1i" name="post[written_on(1i)]">\n}
