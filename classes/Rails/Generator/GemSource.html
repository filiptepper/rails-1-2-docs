def each
  Gem::cache.search(%r_generator$/).inject({}) { |latest, gem|
    hem = latest[gem.name]
    latest[gem.name] = gem if hem.nil? or gem.version > hem.version
    latest
  }.values.each { |gem|
    yield Spec.new(gem.name.sub(%r_generator$/, ''), gem.full_gem_path, label)
  }
end
