def inspect
  header = %x#{OPTIONS[:ps] % 1}`.split("\n")[0] + "\n"
  lines  = pids.collect { |pid| %x#{OPTIONS[:ps] % pid}`.split("\n")[1] }
  
  puts(header + lines.join("\n"))
end
