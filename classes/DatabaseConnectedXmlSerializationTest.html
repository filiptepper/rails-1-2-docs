def test_passing_hash_shouldnt_reuse_builder
  options = {:include=>:posts}
  david = authors(:david)
  first_xml_size = david.to_xml(options).size
  second_xml_size = david.to_xml(options).size
  assert_equal first_xml_size, second_xml_size
