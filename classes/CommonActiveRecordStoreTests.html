def test_basics
  s = session_class.new(:session_id => '1234', :data => { 'foo' => 'bar' })
  assert_equal 'bar', s.data['foo']
  assert s.save
  assert_equal 'bar', s.data['foo']

  assert_not_nil t = session_class.find_by_session_id('1234')
  assert_not_nil t.data
  assert_equal 'bar', t.data['foo']
end
