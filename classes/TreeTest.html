def test_insert

  assert @extra

  assert_equal @extra.parent, mixins(:tree_1)

  assert_equal 3, mixins(:tree_1).children.size
  assert mixins(:tree_1).children.include?(@extra)
  assert mixins(:tree_1).children.include?(mixins(:tree_2))
