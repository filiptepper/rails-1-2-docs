def initialize(environment)
  @env = environment
  @handle_type = SQL_HANDLE_DBC
  rc, @handle = SQLAllocHandle(@handle_type, @env.handle)
  check_rc(rc)
end

def set_connect_attr(attr, value)

  check_rc(SQLSetConnectAttr(@handle, attr, value))
