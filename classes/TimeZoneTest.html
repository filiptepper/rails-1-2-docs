def test_zone_compare
  zone1 = TimeZone.create( "Test1", 4200 )
  zone2 = TimeZone.create( "Test1", 5600 )
  assert zone1 < zone2
  assert zone2 > zone1

  zone1 = TimeZone.create( "Able", 10000 )
  zone2 = TimeZone.create( "Zone", 10000 )
  assert zone1 < zone2
  assert zone2 > zone1

  zone1 = TimeZone.create( "Able", 10000 )
  assert zone1 == zone1
end
