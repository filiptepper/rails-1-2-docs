def block_format(text)
  formatted = text.split(%r\n\r\n/).collect { |paragraph| 
    Text::Format.new(
      :columns => 72, :first_indent => 2, :body_indent => 2, :text => paragraph
    ).format
  }.join("\n")
  
  # Make list points stand on their own line
  formatted.gsub!(%r[ ]*([*]+) ([^*]*)/) { |s| "  #{$1} #{$2.strip}\n" }
  formatted.gsub!(%r[ ]*([#]+) ([^#]*)/) { |s| "  #{$1} #{$2.strip}\n" }

  formatted
end
