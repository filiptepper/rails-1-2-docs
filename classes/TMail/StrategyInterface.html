def create_dest( obj )
  case obj
  when nil
    StringOutput.new
  when String
    StringOutput.new(obj)
  when IO, StringOutput
    obj
  else
    raise TypeError, 'cannot handle this type of object for dest'
  end
end

def accept_strategy( klass, eol, charset, dest = nil )
  dest ||= ''
  accept klass.new(create_dest(dest), charset, eol)
  dest
end
