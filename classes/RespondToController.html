def custom_constant_handling
  Mime::Type.register("text/x-mobile", :mobile)

  respond_to do |type|
    type.html   { render :text => "HTML"   }
    type.mobile { render :text => "Mobile" }
  end
  
  Mime.send :remove_const, :MOBILE
end
