def test_link_to_unless_current_shows_link
  with_restful_routing do
    get :show, :id => 1
    assert_equal "<a href=\"/tasks\">tasks</a>\n" +
      "<a href=\"#{@request.protocol}#{@request.host_with_port}/tasks\">tasks</a>",
      @response.body
  end
end

def with_restful_routing
  with_routing do |set|
    set.draw do |map|
      map.resources :tasks
    end
    yield
  end
end
