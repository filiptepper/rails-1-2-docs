def initialize(seed1, seed2)
  @max_value = 0x3FFFFFFF
  @seed1 = seed1 % @max_value
  @seed2 = seed2 % @max_value
end

def rnd()
  @seed1 = (@seed1*3+@seed2) % @max_value
  @seed2 = (@seed1+@seed2+33) % @max_value
  @seed1.to_f / @max_value
end
