def test_reordering
  assert_equal [mixins(:list_1),
                mixins(:list_2),
                mixins(:list_3),
                mixins(:list_4)],
                ListMixin.find(:all, :conditions => 'parent_id = 5', :order => 'pos')

  mixins(:list_2).move_lower

  assert_equal [mixins(:list_1),
                mixins(:list_3),
                mixins(:list_2),
                mixins(:list_4)],
                ListMixin.find(:all, :conditions => 'parent_id = 5', :order => 'pos')

  mixins(:list_2).move_higher

  assert_equal [mixins(:list_1),
                mixins(:list_2),
                mixins(:list_3),
                mixins(:list_4)],
                ListMixin.find(:all, :conditions => 'parent_id = 5', :order => 'pos')

  mixins(:list_1).move_to_bottom

  assert_equal [mixins(:list_2),
                mixins(:list_3),
                mixins(:list_4),
                mixins(:list_1)],
                ListMixin.find(:all, :conditions => 'parent_id = 5', :order => 'pos')

  mixins(:list_1).move_to_top

  assert_equal [mixins(:list_1),
                mixins(:list_2),
                mixins(:list_3),
                mixins(:list_4)],
                ListMixin.find(:all, :conditions => 'parent_id = 5', :order => 'pos')


  mixins(:list_2).move_to_bottom

  assert_equal [mixins(:list_1),
                mixins(:list_3),
                mixins(:list_4),
                mixins(:list_2)],
                ListMixin.find(:all, :conditions => 'parent_id = 5', :order => 'pos')

  mixins(:list_4).move_to_top

  assert_equal [mixins(:list_4),
                mixins(:list_1),
                mixins(:list_3),
                mixins(:list_2)],
                ListMixin.find(:all, :conditions => 'parent_id = 5', :order => 'pos')

end
