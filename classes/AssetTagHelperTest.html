def test_preset_empty_asset_id
  Object.send(:const_set, :RAILS_ROOT, File.dirname(__FILE__) + "/../fixtures/")
  # on windows, setting ENV["XXX"] to "" makes ENV["XXX"] return nil
  if RUBY_PLATFORM =~ %rwin32/
    ENV["RAILS_ASSET_ID"] = " "
  else
  ENV["RAILS_ASSET_ID"] = ""
  end
  assert_equal %Q(<img alt="Rails" src="/images/rails.png" />), image_tag("rails.png")
end

def test_timebased_asset_id
  Object.send(:const_set, :RAILS_ROOT, File.dirname(__FILE__) + "/../fixtures/")
